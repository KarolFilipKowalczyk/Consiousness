\section{Theory Selection Criteria}

\begin{multicols}{2}
\noindent\textbf{Empirical Adequacy}
\begin{itemize}[nosep]
\item Test competing predictions
\item See which confirmed
\item Update credences
\end{itemize}

\noindent\textbf{Explanatory Power}
\begin{itemize}[nosep]
\item Count phenomena explained
\item Assess depth of explanation
\item Consider unification
\end{itemize}

\noindent\textbf{Parsimony}
\begin{itemize}[nosep]
\item Count assumptions
\item Assess necessity of components
\item Prefer simpler if equal power
\end{itemize}

\noindent\textbf{Testability}
\begin{itemize}[nosep]
\item Count testable predictions
\item Assess feasibility of tests
\item Prefer more falsifiable
\end{itemize}

\noindent\textbf{Practical Utility}
\begin{itemize}[nosep]
\item Clinical applications
\item AI development
\item Enhancement potential
\end{itemize}
\end{multicols}

\begin{keyinsight}
Our framework scores highest on all criteria except parsimony (more complex but explains more). The added complexity is justified by dramatically increased explanatory power.
\end{keyinsight}
